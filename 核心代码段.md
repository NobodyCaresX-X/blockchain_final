### User 数据模型

```python
@dataclass
class User:
    username: str                    # 用户名
    password_hash: str               # 密码哈希
    address: str                     # 区块链地址
    nickname: str                    # 用户昵称
    internal_balance: float = 0.0    # 内置余额（用于众筹）
    external_balance: float = 0.0    # 外置钱包余额
    metamask_address: str = ""       # Metamask地址
```

### 用户注册

```python
def register(self, username: str, password: str, nickname: str = "") -> Optional[User]:
    if username in self._users:
        return None
    
    from .service import service
    service._ensure_initialized()
    
    # 创建新账户
    new_account = service.w3.eth.account.create()
    new_address = new_account.address
        
    # 在测试链上为用户账户分配余额（从默认账户转账）
    try:
        transfer_amount = service.w3.to_wei(100, "ether")
        tx_hash = service.w3.eth.send_transaction({
            "from": service.accounts[0],
            "to": new_address,
            "value": transfer_amount
        })
        service.w3.eth.wait_for_transaction_receipt(tx_hash)
    except Exception as e:
        print(f"转账失败: {e}")
    
    user = User(
        username=username,
        password_hash=self._hash_password(password),
        address=new_address,
        nickname=nickname if nickname else username,
        internal_balance=0.0,
        external_balance=external_balance
    )
    
    self._users[username] = user
    self._address_to_user[new_address.lower()] = user
    
    return user
```

### 用户登录

```python
def login(self, username: str, password: str) -> Optional[str]:
    self._ensure_initialized()
    
    user = self._users.get(username)
    if not user:
        return None
    
    if user.password_hash != self._hash_password(password):
        return None
    
    # 生成JWT令牌
    token = hashlib.sha256(f"{username}{os.urandom(16)}".encode()).hexdigest()
    self._session_tokens[token] = username
    
    return token
```

### 充值功能

```python
def topup(self, username: str, amount: float) -> dict:
    """充值：从外置钱包转入内置余额"""
    self._ensure_initialized()
    user = self._users.get(username)
    if not user:
        return {"success": False, "message": "用户不存在"}
    
    if user.external_balance < amount:
        return {"success": False, "message": f"外置钱包余额不足！当前余额：{user.external_balance:.2f} ETH"}
    
    user.external_balance -= amount
    user.internal_balance += amount
    
    return {
        "success": True,
        "message": f"充值成功！充值金额：{amount} ETH",
        "new_internal_balance": user.internal_balance,
        "new_external_balance": user.external_balance
    }
```

### 退款能力检查

```python
def check_refund_capability(self, username: str, required_amount: float) -> dict:
    """检查退款能力：内置余额不足时，检查外置钱包是否足够"""
    self._ensure_initialized()
    user = self._users.get(username)
    if not user:
        return {"success": False, "message": "用户不存在"}
    
    available = user.internal_balance + user.external_balance
    if available < required_amount:
        return {
            "success": False,
            "message": f"余额不足！需要：{required_amount:.2f} ETH，可用：{available:.2f} ETH（内置：{user.internal_balance:.2f}，外置：{user.external_balance:.2f}）"
        }
    
    return {
        "success": True,
        "internal_balance": user.internal_balance,
        "external_balance": user.external_balance,
        "total_available": available
    }
```

```python
def get_metamask_nonce(self, address: str) -> dict:
    """获取Metamask登录随机数"""
    self._ensure_initialized()
    address = address.lower()
    nonce = os.urandom(32).hex()
    timestamp = int(time.time())
    self._metamask_nonces[address] = {
        "nonce": nonce,
        "timestamp": timestamp
    }
    return {"nonce": nonce, "timestamp": timestamp}
```

### 验证Metamask签名

```python
def verify_metamask_signature(self, address: str, signature: str) -> bool:
    """验证Metamask签名"""
    self._ensure_initialized()
    address = address.lower()
    
    if address not in self._metamask_nonces:
        return False
    
    nonce_data = self._metamask_nonces[address]
    message = f"请签名以登录区块链众筹系统\n地址: {address}\n随机数: {nonce_data['nonce']}\n时间戳: {nonce_data['timestamp']}"
    
    try:
        from web3 import Web3
        w3 = Web3()
        recovered_address = w3.eth.account.recover_message(
            w3.toBytes(text=message),
            signature=signature
        ).lower()
        
        if recovered_address == address:
            del self._metamask_nonces[address]
            return True
    except Exception as e:
        print(f"验证签名失败: {e}")
    
    return False
```

```python
def _deploy_contract(self, artifact: dict) -> None:
    """在测试链上部署合约"""
    print("正在部署合约到测试链...")
    accounts = self.w3.eth.accounts
    deployer = accounts[0]
    
    bytecode = artifact.get("bytecode", "")
    if not bytecode:
        bytecode = artifact.get("bytecode", "")
    
    Crowdfunding = self.w3.eth.contract(abi=artifact["abi"], bytecode=bytecode)
    tx_hash = Crowdfunding.constructor().transact({"from": deployer})
    tx_receipt = self.w3.eth.wait_for_transaction_receipt(tx_hash)
    self.contract_address = tx_receipt.contractAddress
    self.contract = self.w3.eth.contract(address=self.contract_address, abi=artifact["abi"])
    self.chain_id = self.w3.eth.chain_id
    self.accounts = [Web3.to_checksum_address(account) for account in accounts]
    self.ready = True
    print(f"合约已部署到: {self.contract_address}")
```

```python
    projects = []
```

### 格式化项目数据

```python
def _format_project(self, project_id: int) -> ProjectSnapshot:
    self._ensure_initialized()
    self.ensure_ready()
    data = self.contract.functions.getProject(project_id).call()
    donors = self.contract.functions.getDonors(project_id).call()
    
    stages = []
    stage_count = _to_int(data[16])
    for index in range(stage_count):
        stage = self.contract.functions.getStage(project_id, index).call()
        stages.append(
            {
                "index": index,
                "description": stage[0],
                "completionThresholdBps": _to_int(stage[1]),
                "completionThresholdPercent": _to_int(stage[1]) / 100,
                "releaseBps": _to_int(stage[2]),
                "releasePercent": _to_int(stage[2]) / 100,
                "completed": bool(stage[3]),
                "released": bool(stage[4]),
            }
        )
    
    rewards = []
    reward_count = _to_int(data[17])
    for index in range(reward_count):
        reward = self.contract.functions.getReward(project_id, index).call()
        rewards.append(
            {
                "index": index,
                "fundingThresholdWei": _to_int(reward[0]),
                "fundingThresholdEth": _wei_to_eth(_to_int(reward[0])),
                "promise": reward[1],
                "expectedMonth": _to_int(reward[2]),
                "backersCount": _to_int(reward[3]),
            }
        )

    donor_rows = []
    for donor in donors:
        donation = self.contract.functions.getDonation(project_id, donor).call()
        is_monthly = bool(self.contract.functions.isMonthlySupporter(project_id, donor).call())
        donor_rows.append(
            {
                "address": Web3.to_checksum_address(donor),
                "donationWei": _to_int(donation),
                "donationEth": _wei_to_eth(_to_int(donation)),
                "earlySupporter": bool(self.contract.functions.isEarlySupporter(project_id, donor).call()),
                "monthlySupporter": is_monthly,
            }
        )

    deadline = _to_int(data[5])
    chain_time = self.w3.eth.get_block('latest').timestamp

    return ProjectSnapshot(
        id=_to_int(data[0]),
        creator=Web3.to_checksum_address(data[1]),
        name=data[2],
        description=data[3],
        goalWei=_to_int(data[4]),
        goalEth=_wei_to_eth(_to_int(data[4])),
        deadline=deadline,
        deadlineIso=_iso_from_timestamp(deadline),
        pledgedWei=_to_int(data[6]),
        pledgedEth=_wei_to_eth(_to_int(data[6])),
        balanceWei=_to_int(data[7]),
        balanceEth=_wei_to_eth(_to_int(data[7])),
        totalWithdrawnWei=_to_int(data[8]),
        totalWithdrawnEth=_wei_to_eth(_to_int(data[8])),
        totalRefundedWei=_to_int(data[9]),
        totalRefundedEth=_wei_to_eth(_to_int(data[9])),
        status=self._status_label(_to_int(data[10])),
        ended=bool(data[11]),
        creatorWithdrawn=bool(data[12]),
        donorCount=_to_int(data[13]),
        hasStages=bool(data[14]),
        hasMonthlySupport=bool(data[15]),
        stageCount=_to_int(data[16]),
        rewardCount=_to_int(data[17]),
        timeLeftSeconds=max(0, deadline - chain_time),
        donors=donor_rows,
        stages=stages,
        rewards=rewards,
    )
```

### 阶段点自动保留逻辑（合约）

```solidity
function _calculateProportionalRefund(uint256 projectId) private {
    Stage[] storage stages = projectStages[projectId];
    Project storage project = projects[projectId];

    uint256 autoRetained = 0;
    
    for (uint256 i = 0; i < stages.length; i++) {
        uint256 requiredPledged = (project.goal * stages[i].completionThresholdBps) / BPS;
        if (project.pledged >= requiredPledged) {
            uint256 releaseAmount = (project.goal * stages[i].releaseBps) / BPS;
            autoRetained += releaseAmount;
            
            if (!stages[i].completed) {
                stages[i].completed = true;
            }
            if (!stages[i].released) {
                stages[i].released = true;
            }
        }
    }

    project.totalWithdrawn = autoRetained;
}
```

```python
@app.post("/api/projects/{project_id}/refund")
async def refund(project_id: int, request: Request):
    token = request.headers.get("Authorization", "").replace("Bearer ", "")
    if not token:
        raise HTTPException(status_code=401, detail="未登录")
    
    user = auth_service.validate_token(token)
    if not user:
        raise HTTPException(status_code=401, detail="登录已过期")
    
    # 获取项目信息
    project = service.get_project(project_id)
    
    # 查找用户的捐赠金额
    user_donation = None
    for donor in project["donors"]:
        if donor["address"].lower() == user.address.lower():
            user_donation = donor
            break
    
    if not user_donation:
        raise HTTPException(status_code=400, detail="您未向该项目捐赠")
    
    refund_amount = user_donation["donationEth"]
    
    # 检查退款能力
    capability = auth_service.check_refund_capability(user.username, refund_amount)
    if not capability["success"]:
        raise HTTPException(status_code=400, detail=capability["message"])
    
    # 优先使用内置余额
    if user.internal_balance >= refund_amount:
        # 内置余额足够，直接退款
        pass
    else:
        # 内置余额不足，从外置钱包扣除差额
        external_needed = refund_amount - user.internal_balance
        if user.external_balance < external_needed:
            raise HTTPException(status_code=400, detail="外置钱包余额不足")
        user.external_balance -= external_needed
        user.internal_balance = 0
    
    # 调用智能合约退款
    result = service.refund(project_id, user.address)
    
    return {
        "message": "退款成功！",
        "txHash": result["txHash"],
        "refund_amount": refund_amount
    }
```

```python
async def release_milestone(project_id: int, index: int, request: Request):

    if not token:
        raise HTTPException(status_code=401, detail="未登录")

        raise HTTPException(status_code=401, detail="登录已过期")

    # 验证用户是创建者
    if project["creator"].lower() != user.address.lower():
        raise HTTPException(status_code=403, detail="只有项目创建者可以释放里程碑资金")
    
    # 获取里程碑信息
    if index >= len(project["milestones"]):
        raise HTTPException(status_code=400, detail="里程碑索引无效")
    
    milestone = project["milestones"][index]
    
    if not milestone["completed"]:
        raise HTTPException(status_code=400, detail="里程碑尚未完成")
    
    if milestone["released"]:
        raise HTTPException(status_code=400, detail="里程碑资金已释放")
    
    # 计算释放金额
    release_amount = project["pledgedEth"] * (milestone["releaseBps"] / 10000)
    
    # 调用智能合约释放资金
    result = service.release_milestone(project_id, index, user.address)
    
    # 增加创建者内置余额
    auth_service.add_to_internal(user.username, release_amount)
    
    return {
        "message": "里程碑资金释放成功！",
        "txHash": result["txHash"],
        "release_amount": release_amount
    }
```

```python
def _sync_chain_time(self) -> None:
    """同步链上时间与系统时间（初始化时调用）
    
    只用 evm_mine 出块，不用 evm_increaseTime（避免时间超前问题）
    - 链上时间落后系统时间：evm_mine 自动用系统时间出块
    - 链上时间超前系统时间：无法调整（Hardhat不支持回拨）
    """
    chain_time = self.w3.eth.get_block('latest').timestamp
    system_time = int(datetime.now(tz=timezone.utc).timestamp())
    time_diff = chain_time - system_time
    
    if time_diff < -5:
        self.w3.provider.make_request("evm_mine", [])
        new_chain_time = self.w3.eth.get_block('latest').timestamp
        print(f"[TIME_SYNC] 初始化同步: 链上{chain_time} -> {new_chain_time}")
    elif time_diff > 60:
        print(f"[TIME_SYNC] 警告：链上时间超前系统时间{time_diff}秒")
```

```solidity
function refund(uint256 projectId) external projectExists(projectId) {
    Project storage project = projects[projectId];
    require(project.status == ProjectStatus.Failed, "project not failed");

    uint256 donationAmount = donations[projectId][msg.sender];
    require(donationAmount > 0, "nothing to refund");
    require(!donatedRefunded[projectId][msg.sender], "already refunded");

    uint256 totalPledged = project.pledged;
    uint256 totalWithdrawn = project.totalWithdrawn;
    
    uint256 refundBps = 0;
    if (totalPledged > totalWithdrawn) {
        refundBps = ((totalPledged - totalWithdrawn) * BPS) / totalPledged;
    }
    
    uint256 refundAmount = (donationAmount * refundBps) / BPS;
    require(refundAmount > 0, "nothing to refund after stage release");
    require(project.balance >= refundAmount, "insufficient project balance");

    project.totalRefunded += refundAmount;
    project.balance -= refundAmount;
    donatedRefunded[projectId][msg.sender] = true;

    (bool success, ) = payable(msg.sender).call{value: refundAmount}("");
    require(success, "refund failed");

    emit DonationRefunded(projectId, msg.sender, refundAmount);
}
```

```javascript
async function signMessage(message) {
  if (!state.metamaskAddress || !window.ethereum) {
    throw new Error("请先连接Metamask");
  }

  // 使用原生方法将消息转换为十六进制
  const msgHex = "0x" + Array.from(new TextEncoder().encode(message))
    .map(b => b.toString(16).padStart(2, '0'))
    .join('');

  const signature = await window.ethereum.request({
    method: "personal_sign",
    params: [msgHex, state.metamaskAddress],
  });
  return signature;
}
```

```python
def _transact(self, function_call: Any, sender: str, value_wei: int = 0) -> str:
    self._ensure_initialized()
    self.ensure_ready()
    sender = Web3.to_checksum_address(sender)
    tx_options: dict[str, Any] = {"from": sender}
    if value_wei > 0:
        tx_options["value"] = int(value_wei)
    tx_hash = function_call.transact(tx_options)
    receipt = self.w3.eth.wait_for_transaction_receipt(tx_hash)
    return receipt.transactionHash.hex()
```

## 总结

本系统实现了完整的区块链众筹功能，包括：

1. **双钱包系统** - 内置余额用于众筹，外置钱包用于充值和提现
2. **用户认证** - 支持账号密码和Metamask签名两种登录方式
3. **众筹管理** - 创建项目、捐赠、自动结束、里程碑管理
4. **退款机制** - 优先使用内置余额，不足时自动从外置钱包扣除
5. **Metamask集成** - 连接、签名、地址关联、一键登录

系统采用前后端分离架构，后端使用FastAPI提供RESTful API，前端使用原生JavaScript实现动态交互，智能合约使用Solidity编写并部署到区块链上。
